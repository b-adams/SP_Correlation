00866 void  modifyListSortify(List* theData)
00867 {
00868     assert(theData!=NULL);

00884 void  heapSortSiftDown(List* theData, int root, int bottom)
00885 {
00886     assert(theData!=NULL);
00887     
00888     //This process disorders things
00889     theData->isSorted = false;
00890 
00891     int farChild = root * 2 + 1;
00892     int sibling;
00893     
00894     while(farChild <= bottom)
00895     {
00896         // Find the biggest child
00897         if(farChild < bottom) 
00898         {
00899             sibling = farChild + 1;
00900             // Reversed for stability
00901             if(!inOrder(theData->data[farChild], theData->data[sibling]))
00902             {
00903                 farChild = sibling;
00904             }
00905         }
00906         
00907         // If we have the correct ordering, we are done.
00908         if(inOrder(theData->data[root], theData->data[farChild])) return;
00909         
00910         // Swap
00911         swapF(&theData->data[root], &theData->data[farChild]);
00912         
00913         root=farChild;
00914         farChild = root * 2 + 1;
00915     }
00916 }

00939     //Binary search for a match
00940     int iguess = (theData->count)/2; //Search index
00941     int ilo=0;  //Low index
00942     int ihi=theData->count; //High index
00943     float candidate = theData->data[iguess]; //Found data
00944     
00945     while(candidate != datum) 
00946     {
00947         //DELETEME printf("Datum(%f)!=Candidate(%f). <%d | %d | %d>\n", datum, candidate, ilo, iguess, ihi);

00950             theRank->count = 0; 
00951             theRank->rank = (ihi+ilo)/2.0;
00952             return theRank;
00953         }
00954         //DELETEME printf("DBEUG: Check this faster search work!\n");
00955         if(inOrder(candidate, datum)) ihi=iguess-1;
00956         else ilo=iguess+1;
00957         iguess = (ilo+ihi)/2;
00958         candidate = theData->data[iguess];
00959     }
00960     
00961     ilo=ihi=iguess; //Expand match to range
00962     while(theData->data[ilo]==datum && ilo>=0) ilo--; //find pre-firt-match

00964

00966     theRank->count = ((ihi-1)-ilo); //Elements between low and high
00967     theRank->rank = (ilo+ihi)/2.0; //Average position btwn low and high
00968     theRank->indexEnd = ihi-1;
